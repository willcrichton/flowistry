42impl<'tcx> Visitor<'tcx> for GatherBorrows<'tcx> {
43  fn visit_assign(
44    &mut self,
45    _place: &Place<'tcx>,
46    rvalue: &Rvalue<'tcx>,
47    _location: Location,
48  ) {
49    if let Rvalue::Ref(region_pat!(region), kind, borrowed_place) = rvalue {
50      self.borrows.push((*region, *kind, *borrowed_place));
51    }
52  }

72  /// Runs the alias analysis on a given `body_with_facts`.
73  pub fn build(
74    tcx: TyCtxt<'tcx>,
75    def_id: DefId,
76    body_with_facts: &'a BodyWithBorrowckFacts<'tcx>,
77  ) -> Self {
78    let loans = Self::compute_loans(tcx, def_id, body_with_facts, |_, _, _| true);
79    Aliases {
80      tcx,
81      body: &body_with_facts.body,
82      loans,
83    }
84  }

86  /// Alternative constructor if you need to filter out certain borrowck facts.
87  ///
88  /// Just use [`Aliases::build`] unless you know what you're doing.
89  pub fn build_with_fact_selection(
90    tcx: TyCtxt<'tcx>,
91    def_id: DefId,
92    body_with_facts: &'a BodyWithBorrowckFacts<'tcx>,
93    selector: impl Fn(RegionVid, RegionVid, BorrowckLocationIndex) -> bool,
94  ) -> Self {
95    let loans = Self::compute_loans(tcx, def_id, body_with_facts, selector);
96    Aliases {
97      tcx,
98      body: &body_with_facts.body,
99      loans,
100    }

103  fn compute_loans(
104    tcx: TyCtxt<'tcx>,
105    def_id: DefId,
106    body_with_facts: &'a BodyWithBorrowckFacts<'tcx>,
107    constraint_selector: impl Fn(RegionVid, RegionVid, BorrowckLocationIndex) -> bool,
108  ) -> LoanMap<'tcx> {
109    let start = Instant::now();
110    let body = &body_with_facts.body;
111    let static_region = RegionVid::from_usize(0);
112    let input_facts = &body_with_facts.input_facts.as_ref().unwrap();
113    let subset_base = input_facts
114      .subset_base
115      .iter()
116      .cloned()
117      .map(|(r1, r2, i)| (RegionVid::from(r1), RegionVid::from(r2), i))
118      .filter(|(r1, r2, i)| constraint_selector(*r1, *r2, *i))
119      .collect::<Vec<_>>();
120
121    let all_pointers = body
122      .local_decls()
123      .indices()
124      .flat_map(|local| {
125        Place::from_local(local, tcx).interior_pointers(tcx, body, def_id)
126      })
127      .collect::<Vec<_>>();
128    let max_region = all_pointers
129      .iter()
130      .map(|(region, _)| *region)
131      .chain(subset_base.iter().flat_map(|(r1, r2, _)| [*r1, *r2]))
132      .filter(|r| *r != UNKNOWN_REGION)
133      .max()
134      .unwrap_or(static_region);
135    let num_regions = max_region.as_usize() + 1;
136    let all_regions = (0 .. num_regions).map(RegionVid::from_usize);
137
138    let mut subset = SparseBitMatrix::new(num_regions);
139
140    let async_hack = AsyncHack::new(tcx, body, def_id);
141    let ignore_regions = async_hack.ignore_regions();
142
143    // subset('a, 'b) :- subset_base('a, 'b, _).
144    for (a, b, _) in subset_base {
145      if ignore_regions.contains(&a) || ignore_regions.contains(&b) {
146        continue;
147      }
148      subset.insert(a, b);
149    }
150
151    // subset('static, 'a).
152    for a in all_regions.clone() {
153      subset.insert(static_region, a);
154    }
155
156    if is_extension_active(|mode| mode.pointer_mode == PointerMode::Conservative) {
157      // for all p1 : &'a T, p2: &'b T: subset('a, 'b).
158      let mut region_to_pointers: HashMap<_, Vec<_>> = HashMap::default();
159      for (region, places) in &all_pointers {
160        if *region != UNKNOWN_REGION {
161          region_to_pointers
162            .entry(*region)
163            .or_default()
164            .extend(places);
165        }
166      }
167
168      let constraints = generate_conservative_constraints(
169        tcx,
170        &body_with_facts.body,
171        &region_to_pointers,
172      );
173
174      for (a, b) in constraints {
175        subset.insert(a, b);
176      }
177    }
178
179    let mut contains: LoanMap<'tcx> = HashMap::default();
180    let mut definite: HashMap<RegionVid, (Ty<'tcx>, Vec<PlaceElem<'tcx>>)> =
181      HashMap::default();
182
183    // For all e = &'a x.q in body:
184    //   contains('a, p).
185    //   If p = p^[* p2]: definite('a, ty(p2), p2^[])
186    //   Else:            definite('a, ty(p),  p^[]).
187    let mut gather_borrows = GatherBorrows::default();
188    gather_borrows.visit_body(body);
189    for (region, kind, place) in gather_borrows.borrows {
190      if place.is_direct(body, tcx) {
191        contains
192          .entry(region)
193          .or_default()
194          .insert((place, kind.to_mutbl_lossy()));
195      }
196
197      let def = match place.refs_in_projection(body, tcx).next() {
198        Some((ptr, proj)) => {
199          let ptr_ty = ptr.ty(body.local_decls(), tcx).ty;
200          (ptr_ty.builtin_deref(true).unwrap(), proj.to_vec())
201        }
202        None => (
203          body.local_decls()[place.local].ty,
204          place.projection.to_vec(),
205        ),
206      };
207      definite.insert(region, def);
208    }
209
210    // For all args p : &'a ω T where 'a is abstract: contains('a, *p, ω).
211    for arg in body.args_iter() {
212      for (region, places) in
213        Place::from_local(arg, tcx).interior_pointers(tcx, body, def_id)
214      {
215        let region_contains = contains.entry(region).or_default();
216        for (place, mutability) in places {
217          // WARNING / TODO: this is a huge hack (that is conjoined w/ all_args).
218          // Need a way to limit the number of possible pointers for functions with
219          // many pointers in the input. This is almost certainly not sound, but hopefully
220          // it works for most cases.
221          if place.projection.len() <= 2 {
222            region_contains.insert((tcx.mk_place_deref(place), mutability));
223          }
224        }
225      }
226    }
227
228    // For all places p : *T or p : Box<T>: contains('UNK, *p, mut).
229    let unk_contains = contains.entry(UNKNOWN_REGION).or_default();
230    for (region, places) in &all_pointers {
231      if *region == UNKNOWN_REGION {
232        for (place, _) in places {
233          unk_contains.insert((tcx.mk_place_deref(*place), Mutability::Mut));
234        }
235      }
236    }
237
238    info!(
239      "Initial places in loan set: {}, total regions {}, definite regions: {}",
240      contains.values().map(|set| set.len()).sum::<usize>(),
241      contains.len(),
242      definite.len()
243    );
244
245    debug!("Initial contains: {contains:#?}");
246    debug!("Definite: {definite:#?}");
247
248    // Compute a topological sort of the subset relation.
249    let edge_pairs = subset
250      .rows()
251      .flat_map(|r1| subset.iter(r1).map(move |r2| (r1, r2)))
252      .collect::<Vec<_>>();
253    let subset_graph = VecGraph::<_, false>::new(num_regions, edge_pairs);
254    let subset_sccs = Sccs::<RegionVid, RegionSccIndex>::new(&subset_graph);
255    let mut scc_to_regions = IndexVec::from_elem_n(
256      ChunkedBitSet::new_empty(num_regions),
257      subset_sccs.num_sccs(),
258    );
259    for r in all_regions.clone() {
260      let scc = subset_sccs.scc(r);
261      scc_to_regions[scc].insert(r);
262    }
263    let scc_order = reverse_post_order(&subset_sccs, subset_sccs.scc(static_region));
264    elapsed("relation construction", start);
265
266    // Subset implies containment: l ∈ 'a ∧ 'a ⊆ 'b ⇒ l ∈ 'b
267    // i.e. contains('b, l) :- contains('a, l), subset('a, 'b).
268    //
269    // contains('b, p2^[p], ω) :-
270    //   contains('a, p, ω), subset('a, 'b),
271    //   definite('b, T, p2^[]), !subset('b, 'a), p : T.
272    //
273    // If 'a is from a borrow expression &'a proj[*p'], then we add proj to all inherited aliases.
274    // See interprocedural_field_independence for an example where this matters.
275    // But we only do this if:
276    //   * !subset('b, 'a) since otherwise projections would be added infinitely.
277    //   * if p' : &T, then p : T since otherwise proj[p] is not well-typed.
278    //
279    // Note that it's theoretically more efficient to compute the transitive closure of `subset`
280    // and then do the pass below in one step rather than to a fixpoint. But this negates the added
281    // precision from propagating projections. For example, in the program:
282    //   a = &'0 mut (0, 0)
283    //   b = &'1 mut a.0
284    //   c = &'2 mut *b
285    //   *c = 1;
286    // then '0 :> '1 :> '2. By propagating projections, then '1 = {a.0}. However if we see '0 :> '2
287    // to insert contains('0) into contains('2), then contains('2) = {a, a.0} which defeats the purpose!
288    // Then *c = 1 is considered to be a mutation to anything within a.
289    //
290    // Rather than iterating over the entire subset relation, we only do local fixpoints
291    // within each strongly-connected component.
292    let start = Instant::now();
293    for r in all_regions {
294      contains.entry(r).or_default();
295    }
296    for scc_idx in scc_order {
297      loop {
298        let mut changed = false;
299        let scc = &scc_to_regions[scc_idx];
300        for a in scc.iter() {
301          for b in subset.iter(a) {
302            if a == b {
303              continue;
304            }
305
306            // SAFETY: a != b
307            let a_contains =
308              unsafe { &*(contains.get(&a).unwrap() as *const LoanSet<'tcx>) };
309            let b_contains =
310              unsafe { &mut *(contains.get_mut(&b).unwrap() as *mut LoanSet<'tcx>) };
311
312            let cyclic = scc.contains(b);
313            match definite.get(&b) {
314              Some((ty, proj)) if !cyclic => {
315                for (p, mutability) in a_contains.iter() {
316                  let p_ty = p.ty(body.local_decls(), tcx).ty;
317                  let p_proj = if *ty == p_ty {
318                    let mut full_proj = p.projection.to_vec();
319                    full_proj.extend(proj);
320                    Place::make(p.local, tcx.mk_place_elems(&full_proj), tcx)
321                  } else {
322                    *p
323                  };
324
325                  changed |= b_contains.insert((p_proj, *mutability));
326                }
327              }
328              _ => {
329                let orig_len = b_contains.len();
330                b_contains.extend(a_contains);
331                changed |= b_contains.len() != orig_len;
332              }
333            }
334          }
335        }
336
337        if !changed {
338          break;
339        }
340      }
341    }
342    elapsed("fixpoint", start);
343
344    info!(
345      "Final places in loan set: {}",
346      contains.values().map(|set| set.len()).sum::<usize>()
347    );
348
349    log::trace!("contains: {contains:#?}");
350
351    contains
352  }
353
354  /// Given a `place`, returns the set of direct places it could refer to.
355  ///
356  /// For example, in the program:
357  /// ```
358  /// let x = 1;
359  /// let y = &x;
360  /// ```
361  ///
362  /// The place `*y` (but NOT `y`) is an alias for `x`. Similarly, in the program:
363  ///
364  /// ```
365  /// let v = vec![1, 2, 3];
366  /// let n = &v[0];
367  /// ```
368  ///
369  /// The place `*n` is an alias for `v` (even though they have different types!).
370  pub fn aliases(&self, place: Place<'tcx>) -> PlaceSet<'tcx> {
371    let mut aliases = HashSet::default();
372    aliases.insert(place);
373
374    // Places with no derefs, or derefs from arguments, have no aliases
375    if place.is_direct(self.body, self.tcx) {
376      return aliases;
377    }
378
379    // place = after[*ptr]
380    let (ptr, after) = place
381      .refs_in_projection(self.body, self.tcx)
382      .last()
383      .unwrap();
384
385    // ptr : &'region orig_ty
386    let ptr_ty = ptr.ty(self.body.local_decls(), self.tcx).ty;
387    let (region, orig_ty) = match ptr_ty.kind() {
388      _ if ptr_ty.is_box() => (
389        UNKNOWN_REGION,
390        ptr_ty.boxed_ty().expect("Could not unbox boxed type??"),
391      ),
392      TyKind::RawPtr(ty, _) => (UNKNOWN_REGION, *ty),
393      TyKind::Ref(Region(Interned(RegionKind::ReVar(region), _)), ty, _) => {
394        (*region, *ty)
395      }
396      _ => return aliases,
397    };
398
399    // For each p ∈ loans('region),
400    //   if p : orig_ty then add: after[p]
401    //   else add: p
402    let region_loans = self
403      .loans
404      .get(&region)
405      .map(|loans| loans.iter())
406      .into_iter()
407      .flatten();
408    let region_aliases = region_loans.map(|(loan, _)| {
409      let loan_ty = loan.ty(self.body.local_decls(), self.tcx).ty;
410      if orig_ty == loan_ty {
411        let mut projection = loan.projection.to_vec();
412        projection.extend(after.iter().copied());
413        Place::make(loan.local, &projection, self.tcx)
414      } else {
415        *loan
416      }
417    });
418
419    aliases.extend(region_aliases);
420    log::trace!("Aliases for place {place:?} are {aliases:?}");
421
422    aliases
423  }
424}

426fn generate_conservative_constraints<'tcx>(
427  tcx: TyCtxt<'tcx>,
428  body: &Body<'tcx>,
429  region_to_pointers: &HashMap<RegionVid, Vec<(Place<'tcx>, Mutability)>>,
430) -> Vec<(RegionVid, RegionVid)> {
431  let get_ty = |p| tcx.mk_place_deref(p).ty(body.local_decls(), tcx).ty;
432  let same_ty = |p1, p2| get_ty(p1) == get_ty(p2);
433
434  region_to_pointers
435    .iter()
436    .flat_map(|(region, places)| {
437      let regions_with_place = region_to_pointers
438        .iter()
439        // find other regions that contain a loan matching any type in places
440        .filter(|(other_region, other_places)| {
441          *region != **other_region
442            && places.iter().any(|(place, _)| {
443              other_places
444                .iter()
445                .any(|(other_place, _)| same_ty(*place, *other_place))
446            })
447        });
448
449      // add 'a : 'b and 'b : 'a to ensure the lifetimes are considered equal
450      regions_with_place
451        .flat_map(|(other_region, _)| {
452          [(*region, *other_region), (*other_region, *region)]
453        })
454        .collect::<Vec<_>>()
455    })
456    .collect::<Vec<_>>()
457}

482  #[test]
483  fn test_aliases_basic() {
484    let input = r#"
485    fn main() {
486      fn foo<'a, 'b>(x: &'a i32, y: &'b i32) -> &'a i32 { x }
487
488      let a = 1;
489      let b = 2;
490      let c = &a;
491      let d = &b;
492      let e = foo(c, d);      
493    }
494    "#;
495    alias_harness(input, |tcx, body, aliases| {
496      let p = Placer::new(tcx, body);
497      let d_deref = p.local("d").deref().mk();
498      let e_deref = p.local("e").deref().mk();
499
500      // `*e` aliases only `a` (not `b`) because of the lifetime constraints on `foo`
501      compare_sets(
502        aliases.aliases(e_deref),
503        hashset! { p.local("a").mk(), e_deref },
504      );
505
506      // `*e` aliases only `b` because nothing might relate it to `a`
507      compare_sets(
508        aliases.aliases(d_deref),
509        hashset! { p.local("b").mk(), d_deref },
510      );
511    });
512  }
