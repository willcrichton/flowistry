14/// An unordered collections of MIR [`Place`]s.
15///
16/// *Note:* this used to be implemented as an [`IndexSet`](indexical::IndexSet),
17/// but in practice it was very hard to determine up-front a fixed domain of
18/// [`Place`]s that was not "every possible place in the body".
19pub type PlaceSet<'tcx> = HashSet<Place<'tcx>>;
20
21/// Given the arguments to a function, returns all projections of the arguments that are mutable pointers.
22pub fn arg_mut_ptrs<'tcx>(
23  args: &[(usize, Place<'tcx>)],
24  tcx: TyCtxt<'tcx>,
25  body: &Body<'tcx>,
26  def_id: DefId,
27) -> Vec<(usize, Place<'tcx>)> {
28  let ignore_mut =
29    is_extension_active(|mode| mode.mutability_mode == MutabilityMode::IgnoreMut);
30  args
31    .iter()
32    .flat_map(|(i, place)| {
33      place
34        .interior_pointers(tcx, body, def_id)
35        .into_iter()
36        .flat_map(|(_, places)| {
37          places
38            .into_iter()
39            .filter_map(|(place, mutability)| match mutability {
40              Mutability::Mut => Some(place),
41              Mutability::Not => ignore_mut.then_some(place),
42            })
43        })
44        .map(move |place| (*i, tcx.mk_place_deref(place)))
45    })
46    .collect::<Vec<_>>()
47}

49/// Given the arguments to a function, returns all places in the arguments.
50pub fn arg_places<'tcx>(args: &[Spanned<Operand<'tcx>>]) -> Vec<(usize, Place<'tcx>)> {
51  args
52    .iter()
53    .enumerate()
54    .filter_map(|(i, arg)| arg.node.as_place().map(move |place| (i, place)))
55    .collect::<Vec<_>>()
56}

58/// A hack to temporary hack to reduce spurious dependencies in generators
59/// arising from async functions.
60///
61/// The issue is that the `&mut std::task::Context` variable interferes with both
62/// the modular approximation and the alias analysis. As a patch up, we ignore subset
63/// constraints arising from lifetimes appearing in the Context type, as well as ignore
64/// any place of type Context in function calls.
65///
66/// See test: async_two_await
67pub(crate) struct AsyncHack<'a, 'tcx> {
68  context_ty: Option<Ty<'tcx>>,
69  tcx: TyCtxt<'tcx>,
70  body: &'a Body<'tcx>,
71}
72
73impl<'a, 'tcx> AsyncHack<'a, 'tcx> {
74  pub fn new(tcx: TyCtxt<'tcx>, body: &'a Body<'tcx>, def_id: DefId) -> Self {
75    let context_ty = body.async_context(tcx, def_id);
76    AsyncHack {
77      context_ty,
78      tcx,
79      body,
80    }
81  }
82
83  pub fn ignore_regions(&self) -> HashSet<RegionVid> {
84    match self.context_ty {
85      Some(context_ty) => context_ty
86        .walk()
87        .filter_map(|part| match part.kind() {
88          GenericArgKind::Lifetime(r) => match r.kind() {
89            RegionKind::ReVar(rv) => Some(rv),
90            _ => None,
91          },
92          _ => None,
93        })
94        .collect::<HashSet<_>>(),
95      None => HashSet::default(),
96    }
97  }
98
99  pub fn ignore_place(&self, place: Place<'tcx>) -> bool {
100    match self.context_ty {
101      Some(context_ty) => {
102        self
103          .tcx
104          .erase_regions(place.ty(&self.body.local_decls, self.tcx).ty)
105          == self.tcx.erase_regions(context_ty)
106      }
107      None => false,
108    }
109  }
110}
