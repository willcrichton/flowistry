25/// The output of the information flow analysis.
26///
27/// Using the metavariables in [the paper](https://arxiv.org/abs/2111.13662): for each
28/// [`LocationOrArg`](rustc_utils::mir::location_or_arg::LocationOrArg) $\ell$ in a [`Body`](rustc_middle::mir::Body) $f$,
29/// this type contains a [`FlowDomain`] $\Theta_\ell$ that maps from a [`Place`](rustc_middle::mir::Place) $p$
30/// to a [`LocationOrArgSet`](rustc_utils::mir::location_or_arg::index::LocationOrArgSet) $\kappa$. The domain of $\Theta_\ell$
31/// is all places that have been defined up to $\ell$. For each place, $\Theta_\ell(p)$ contains the set of locations
32/// (or arguments) that could influence the value of that place, i.e. the place's dependencies.
33///
34/// For example, to get the dependencies of the first argument at the first instruction, that would be:
35/// ```
36/// # #![feature(rustc_private)]
37/// # extern crate rustc_middle;
38/// # use rustc_middle::{ty::TyCtxt, mir::{Place, Location, Local}};
39/// # use flowistry::{infoflow::{FlowDomain, FlowResults}};
40/// # use rustc_utils::{mir::location_or_arg::index::LocationOrArgSet, PlaceExt};
41/// fn example<'tcx>(tcx: TyCtxt<'tcx>, results: &FlowResults<'_, 'tcx>) {
42///   let ℓ: Location         = Location::START;
43///   let Θ: &FlowDomain      = results.state_at(ℓ);
44///   let p: Place            = Place::make(Local::from_usize(1), &[], tcx);
45///   let κ: LocationOrArgSet = results.analysis.deps_for(Θ, p);
46///   for ℓ2 in κ.iter() {
47///     println!("at location {ℓ:?}, place {p:?} depends on location {ℓ2:?}");
48///   }
49/// }
50/// ```
51///
52/// To access a [`FlowDomain`] for a given location, use the method [`AnalysisResults::state_at`](engine::AnalysisResults::state_at).
53/// See [`FlowDomain`] for more on how to access the location set for a given place.
54///
55/// **Note:** this analysis uses rustc's [dataflow analysis framework](https://rustc-dev-guide.rust-lang.org/mir/dataflow.html),
56/// i.e. [`rustc_mir_dataflow`](https://doc.rust-lang.org/nightly/nightly-rustc/rustc_mir_dataflow/index.html).
57/// You will see several types and traits from that crate here, such as
58/// [`Analysis`](https://doc.rust-lang.org/nightly/nightly-rustc/rustc_mir_dataflow/trait.Analysis.html) and
59/// [`AnalysisDomain`](https://doc.rust-lang.org/nightly/nightly-rustc/rustc_mir_dataflow/trait.AnalysisDomain.html).
60/// However, for performance purposes, several constructs were reimplemented within Flowistry, such as [`AnalysisResults`](engine::AnalysisResults)
61/// which replaces [`rustc_mir_dataflow::Results`](https://doc.rust-lang.org/nightly/nightly-rustc/rustc_mir_dataflow/struct.Results.html).
62pub type FlowResults<'a, 'tcx> = engine::AnalysisResults<'tcx, FlowAnalysis<'a, 'tcx>>;
63

66    const { RefCell::new(Vec::new()) };

69/// Computes information flow for a MIR body.
70///
71/// See [example.rs](https://github.com/willcrichton/flowistry/tree/master/crates/flowistry/examples/example.rs)
72/// for a complete example of how to call this function.
73///
74/// To get a `BodyWithBorrowckFacts`, you can use the
75/// [`get_body_with_borrowck_facts`](rustc_utils::mir::borrowck_facts::get_body_with_borrowck_facts)
76/// function.
77///
78/// See [`FlowResults`] for an explanation of how to use the return value.
79pub fn compute_flow<'a, 'tcx>(
80  tcx: TyCtxt<'tcx>,
81  body_id: BodyId,
82  body_with_facts: &'a BodyWithBorrowckFacts<'tcx>,
83) -> FlowResults<'a, 'tcx> {
84  BODY_STACK.with(|body_stack| {
85    body_stack.borrow_mut().push(body_id);
86    debug!("{}", body_with_facts.body.to_string(tcx).unwrap());
87
88    let def_id = tcx.hir_body_owner_def_id(body_id).to_def_id();
89    let place_info = PlaceInfo::build(tcx, def_id, body_with_facts);
90    let location_domain = place_info.location_domain().clone();
91
92    let body = &body_with_facts.body;
93
94    let results = {
95      block_timer!("Flow");
96
97      let analysis = FlowAnalysis::new(tcx, def_id, body, place_info);
98      engine::iterate_to_fixpoint(tcx, body, location_domain, analysis)
99      // analysis.into_engine(tcx, body).iterate_to_fixpoint()
100    };
101
102    if log::log_enabled!(log::Level::Info) {
103      let counts = body
104        .all_locations()
105        .flat_map(|loc| {
106          let state = results.state_at(loc);
107          state
108            .rows()
109            .map(|(_, locations)| locations.len())
110            .collect::<Vec<_>>()
111        })
112        .collect::<Vec<_>>();
113
114      let nloc = body.all_locations().count();
115      let np = counts.len();
116      let pavg = np as f64 / (nloc as f64);
117      let nl = counts.into_iter().sum::<usize>();
118      let lavg = nl as f64 / (nloc as f64);
119      log::info!(
120        "Over {nloc} locations, total number of place entries: {np} (avg {pavg:.0}/loc), total size of location sets: {nl} (avg {lavg:.0}/loc)",
121      );
122    }
123
124    if std::env::var("DUMP_MIR").is_ok()
125      && BODY_STACK.with(|body_stack| body_stack.borrow().len() == 1)
126    {
127      todo!()
128      // utils::dump_results(body, &results, def_id, tcx).unwrap();
129    }
130
131    body_stack.borrow_mut().pop();
132
133    results
134  })
