16/// Indicator of certainty about whether a place is being mutated.
17/// Used to determine whether an update should be strong or weak.
18#[derive(Debug)]
19pub enum MutationStatus {
20  /// A place is definitely mutated, e.g. `x = y` definitely mutates `x`.
21  Definitely,
22
23  /// A place is possibly mutated, e.g. `f(&mut x)` possibly mutates `x`.
24  Possibly,
25}
26
27/// Information about a particular mutation.
28#[derive(Debug)]
29pub struct Mutation<'tcx> {
30  /// The place that is being mutated.
31  pub mutated: Place<'tcx>,
32
33  /// The set of inputs to the mutating operation.
34  pub inputs: Vec<Place<'tcx>>,
35
36  /// The certainty of whether the mutation is happening.
37  pub status: MutationStatus,
38}
39
40/// MIR visitor that invokes a callback for every [`Mutation`] in the visited object.
41///
42/// Construct the visitor with [`ModularMutationVisitor::new`], then call one of the
43/// MIR [`Visitor`] methods.
44pub struct ModularMutationVisitor<'a, 'tcx, F>
45where
46  F: FnMut(Location, Vec<Mutation<'tcx>>),
47{
48  f: F,
49  place_info: &'a PlaceInfo<'a, 'tcx>,
50}
51
52impl<'a, 'tcx, F> ModularMutationVisitor<'a, 'tcx, F>
53where
54  F: FnMut(Location, Vec<Mutation<'tcx>>),
55{
56  /// Constructs a new visitor.
57  pub fn new(place_info: &'a PlaceInfo<'a, 'tcx>, f: F) -> Self {
58    ModularMutationVisitor { place_info, f }
59  }
60}
61
62impl<'tcx, F> Visitor<'tcx> for ModularMutationVisitor<'_, 'tcx, F>
63where
64  F: FnMut(Location, Vec<Mutation<'tcx>>),
65{
66  fn visit_assign(
67    &mut self,
68    mutated: &Place<'tcx>,
69    rvalue: &Rvalue<'tcx>,
70    location: Location,
71  ) {
72    debug!("Checking {location:?}: {mutated:?} = {rvalue:?}");
73    let body = self.place_info.body;
74    let tcx = self.place_info.tcx;
75
76    match rvalue {
77      // In the case of _1 = aggregate { field1: op1, field2: op2, ... },
78      // then destructure this into a series of mutations like
79      // _1.field1 = op1, _1.field2 = op2, and so on.
80      Rvalue::Aggregate(agg_kind, ops) => {
81        let info = match &**agg_kind {
82          AggregateKind::Adt(def_id, idx, substs, _, _) => {
83            let adt_def = tcx.adt_def(*def_id);
84            let variant = adt_def.variant(*idx);
85            let mutated = match adt_def.adt_kind() {
86              AdtKind::Enum => mutated.project_deeper(
87                &[ProjectionElem::Downcast(Some(variant.name), *idx)],
88                tcx,
89              ),
90              AdtKind::Struct | AdtKind::Union => *mutated,
91            };
92            let fields = variant.fields.iter();
93            let tys = fields
94              .map(|field| field.ty(tcx, substs))
95              .collect::<Vec<_>>();
96            Some((mutated, tys))
97          }
98          AggregateKind::Tuple => {
99            let ty = rvalue.ty(body.local_decls(), tcx);
100            Some((*mutated, ty.tuple_fields().to_vec()))
101          }
102          _ => None,
103        };
104
105        if let Some((mutated, tys)) = info
106          && tys.len() > 0
107        {
108          let fields =
109            tys
110              .into_iter()
111              .enumerate()
112              .zip(ops.iter())
113              .map(|((i, ty), input_op)| {
114                let field = PlaceElem::Field(FieldIdx::from_usize(i), ty);
115                let input_place = input_op.as_place();
116                (mutated.project_deeper(&[field], tcx), input_place)
117              });
118
119          let mutations = fields
120            .map(|(mutated, input)| Mutation {
121              mutated,
122              inputs: input.into_iter().collect::<Vec<_>>(),
123              status: MutationStatus::Definitely,
124            })
125            .collect::<Vec<_>>();
126          (self.f)(location, mutations);
127          return;
128        }
129      }
130
131      // In the case of _1 = _2 where _2 : struct Foo { x: T, y: S, .. },
132      // then destructure this into a series of mutations like
133      // _1.x = _2.x, _1.y = _2.y, and so on.
134      Rvalue::Use(Operand::Move(place) | Operand::Copy(place)) => {
135        let place_ty = place.ty(&body.local_decls, tcx).ty;
136        if let TyKind::Adt(adt_def, substs) = place_ty.kind()
137          && adt_def.is_struct()
138        {
139          let fields = adt_def
140            .all_visible_fields(self.place_info.def_id, self.place_info.tcx)
141            .enumerate()
142            .map(|(i, field_def)| {
143              PlaceElem::Field(FieldIdx::from_usize(i), field_def.ty(tcx, substs))
144            });
145          let mut mutations = fields
146            .map(|field| {
147              let mutated_field = mutated.project_deeper(&[field], tcx);
148              let input_field = place.project_deeper(&[field], tcx);
149              Mutation {
150                mutated: mutated_field,
151                inputs: vec![input_field],
152                status: MutationStatus::Definitely,
153              }
154            })
155            .collect::<Vec<_>>();
156
157          if mutations.is_empty() {
158            mutations.push(Mutation {
159              mutated: *mutated,
160              inputs: vec![*place],
161              status: MutationStatus::Definitely,
162            });
163          }
164          (self.f)(location, mutations);
165          return;
166        }
167      }
168
169      _ => {}
170    }
171
172    let mut collector = PlaceCollector::default();
173    collector.visit_rvalue(rvalue, location);
174    (self.f)(location, vec![Mutation {
175      mutated: *mutated,
176      inputs: collector.0,
177      status: MutationStatus::Definitely,
178    }]);
179  }
180
181  fn visit_terminator(&mut self, terminator: &Terminator<'tcx>, location: Location) {
182    debug!("Checking {location:?}: {:?}", terminator.kind);
183    let tcx = self.place_info.tcx;
184
185    match &terminator.kind {
186      TerminatorKind::Call {
187        /*func,*/ // TODO: deal with func
188        args,
189        destination,
190        ..
191      } => {
192        let async_hack = AsyncHack::new(
193          self.place_info.tcx,
194          self.place_info.body,
195          self.place_info.def_id,
196        );
197        let arg_places = utils::arg_places(args)
198          .into_iter()
199          .map(|(_, place)| place)
200          .filter(|place| !async_hack.ignore_place(*place))
201          .collect::<Vec<_>>();
202        let arg_inputs = arg_places.clone();
203
204        let ret_is_unit = destination
205          .ty(self.place_info.body.local_decls(), tcx)
206          .ty
207          .is_unit();
208        let inputs = if ret_is_unit {
209          Vec::new()
210        } else {
211          arg_inputs.clone()
212        };
213
214        let mut mutations = vec![Mutation {
215          mutated: *destination,
216          inputs,
217          status: MutationStatus::Definitely,
218        }];
219
220        for arg in arg_places {
221          for arg_mut in self.place_info.reachable_values(arg, Mutability::Mut) {
222            mutations.push(Mutation {
223              mutated: *arg_mut,
224              inputs: arg_inputs.clone(),
225              status: MutationStatus::Possibly,
226            });
227          }
228        }
229
230        (self.f)(location, mutations);
231      }
232
233      _ => {}
234    }
235  }
