atty is a simple utility that answers one question

is this a tty?

usage is just as simple

if atty::is(atty::Stream::Stdout) {
  println!("i'm a tty")
}
if atty::isnt(atty::Stream::Stdout) {
  println!("i'm not a tty")
}
